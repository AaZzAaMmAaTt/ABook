# ==================== БЕЗОПАСНОЕ РЕДАКТИРОВАНИЕ СООБЩЕНИЙ ====================
async def safe_edit_message(orig_message: types.Message, text: str, reply_markup=None, parse_mode="Markdown"):
    """Попробует edit_caption (если есть caption у сообщения), иначе edit_text."""
    try:
        if getattr(orig_message, "caption", None) is not None:
            await orig_message.edit_caption(caption=text, reply_markup=reply_markup, parse_mode=parse_mode)
        else:
            await orig_message.edit_text(text, reply_markup=reply_markup, parse_mode=parse_mode)
    except TelegramBadRequest as e:
        logging.warning("TelegramBadRequest при редактировании: %s", e)
        try:
            await orig_message.answer(text, reply_markup=reply_markup, parse_mode=parse_mode)
        except Exception as ex:
            logging.exception("Не удалось отправить fallback-сообщение: %s", ex)

# ==================== ПРОВЕРКА АДМИНА ====================
def is_admin(user_id: int) -> bool:

# ==================== ИНИЦИАЛИЗАЦИЯ БАЗЫ ====================
async def init_db():
    os.makedirs("data", exist_ok=True)
    async with aiosqlite.connect(DB_PATH) as db:
        # Таблица пользователей
        await db.execute("""
        CREATE TABLE IF NOT EXISTS users (
            user_id INTEGER PRIMARY KEY,
            username TEXT,
            first_seen DATE,
            read_books INTEGER DEFAULT 0
        )
        """)

        # Таблица книг
        await db.execute("""
        CREATE TABLE IF NOT EXISTS books (
            id INTEGER PRIMARY KEY AUTOINCREMENT,
            title TEXT,
            description TEXT,
            file_path TEXT,
            cover_path TEXT
        )
        """)

        # Таблица для отслеживания чтения
        await db.execute("""
        CREATE TABLE IF NOT EXISTS reading_sessions (
            user_id INTEGER,
            book_id INTEGER,
            start_time DATETIME,
            UNIQUE(user_id, book_id)
            )
        """)

        # Добавляем столбец total_read_time пользователям (если его нет)
        try:
            await db.execute("ALTER TABLE users ADD COLUMN total_read_time INTEGER DEFAULT 0")
        except Exception:
            # Столбец уже есть — пропускаем
            pass

        await db.commit()

@dp.message(F.from_user.id.in_(ADMIN_IDS))
async def add_book_via_admin(message: types.Message):
    if not is_admin(message.from_user.id):
        return  # Игнорируем сообщения обычных пользователей

    if not message.text or "|" not in message.text:
        return  # Пропускаем сообщения, которые не подходят под формат

    try:
        title, desc, file_path, cover_path = map(str.strip, message.text.split("|"))
    except Exception:
        await message.answer("⚠️ Неверный формат. Используйте:\n`Название | Описание | путь_к_файлу | путь_к_обложке`", parse_mode="Markdown")
        return

    if not os.path.exists(file_path):
        await message.answer("❌ Файл книги не найден.")
        return
    if not os.path.exists(cover_path):
        await message.answer("⚠️ Обложка не найдена, книга добавлена без картинки.")
        cover_path = None

    async with aiosqlite.connect(DB_PATH) as db:
        await db.execute(
            "INSERT INTO books (title, description, file_path, cover_path) VALUES (?, ?, ?, ?)",
            (title, desc, file_path, cover_path)
        )
        await db.commit()

    await message.answer(f"✅ Книга *{title}* добавлена!", parse_mode="Markdown")

@dp.callback_query(F.data.startswith("book_"))
async def show_book(callback: types.CallbackQuery, state: FSMContext):
    book_id = int(callback.data.split("_")[1])
    async with aiosqlite.connect(DB_PATH) as db:
        cur = await db.execute("SELECT title, description, cover_path FROM books WHERE id = ?", (book_id,))
        book = await cur.fetchone()

    if not book:
        await callback.answer("Книга не найдена.")
        return

    title, desc, cover_path = book
    caption = f"📘 *{title}*\n\n{desc}"

    # Удаляем старое сообщение книги, если оно есть
    data = await state.get_data()
    prev_book_message_id = data.get("book_message_id")
    if prev_book_message_id:
        try:
            await bot.delete_message(callback.from_user.id, prev_book_message_id)
        except Exception:
            pass

    # Удаляем сообщение каталога и сохраняем его ID
    catalog_message_id = callback.message.message_id
    await state.update_data(catalog_message_id=catalog_message_id)

    try:
        await callback.message.delete()
    except Exception:
        pass

    # Кнопки под описанием книги (без таймера)
    reply_markup = InlineKeyboardMarkup(inline_keyboard=[
        [InlineKeyboardButton(text="📖 Читать", callback_data=f"read_{book_id}")],
        [InlineKeyboardButton(text="⬅ Назад", callback_data="back_to_catalog")]
    ])

    # Отправляем сообщение с описанием книги
    if cover_path and os.path.exists(cover_path):
        sent = await callback.message.answer_photo(
            photo=FSInputFile(cover_path),
            caption=caption,
            reply_markup=reply_markup,
            parse_mode="Markdown"
        )
    else:
        sent = await callback.message.answer(
            caption,
            reply_markup=reply_markup,
            parse_mode="Markdown"
        )

    # Сохраняем ID сообщения книги и статус
    await state.update_data(book_message_id=sent.message_id, current="book")

# ==================== НАЧАТЬ / ОСТАНОВИТЬ ТАЙМЕР ====================
@dp.callback_query(F.data.startswith("start_timer_"))
async def toggle_timer(callback: types.CallbackQuery):
    user_id = callback.from_user.id
    book_id = int(callback.data.split("_")[2]) if "_" in callback.data else int(callback.data.split("_")[1])

    async with aiosqlite.connect(DB_PATH) as db:
        # Проверяем, есть ли активная сессия чтения
        cur = await db.execute("""
            SELECT start_time FROM reading_sessions 
            WHERE user_id = ? AND book_id = ?
        """, (user_id, book_id))
        active = await cur.fetchone()

        # Если сессия есть → ОСТАНАВЛИВАЕМ ТАЙМЕР
        if active:
            start_time = datetime.fromisoformat(active[0])
            elapsed_seconds = (datetime.now() - start_time).total_seconds()
            elapsed_minutes = int(elapsed_seconds // 60)
            elapsed_text = f"{elapsed_minutes} мин." if elapsed_minutes > 0 else f"{int(elapsed_seconds)} сек."

            # Удаляем запись из reading_sessions
            await db.execute("DELETE FROM reading_sessions WHERE user_id = ? AND book_id = ?", (user_id, book_id))

            # Добавляем время к общему времени пользователя
            await db.execute("""
                UPDATE users 
                SET total_read_time = COALESCE(total_read_time, 0) + ? 
                WHERE user_id = ?
            """, (int(elapsed_seconds), user_id))
            await db.commit()

            # Ответ пользователю
            await callback.answer("⏹ Таймер остановлен.")
            await callback.message.answer(f"📚 Вы читали книгу *{elapsed_text}*", parse_mode="Markdown")

            # Возвращаем кнопку "Начать таймер"
            reply_markup = InlineKeyboardMarkup(inline_keyboard=[
                [InlineKeyboardButton(text="⏱ Начать таймер", callback_data=f"start_timer_{book_id}")],
                [InlineKeyboardButton(text="✅ Книга прочитана", callback_data=f"mark_read_{book_id}")],
                [InlineKeyboardButton(text="⬅ Назад", callback_data="back_to_catalog")]
            ])
            try:
                await callback.message.edit_reply_markup(reply_markup)
            except Exception:
                pass

        # Если сессии нет → ЗАПУСКАЕМ ТАЙМЕР
        else:
            now = datetime.now().isoformat()
            await db.execute("""
                INSERT OR REPLACE INTO reading_sessions (user_id, book_id, start_time)
                VALUES (?, ?, ?)
            """, (user_id, book_id, now))
            await db.commit()

            await callback.answer("⏱ Таймер запущен!")
            # Меняем кнопку на “Остановить таймер”
            reply_markup = InlineKeyboardMarkup(inline_keyboard=[
                [InlineKeyboardButton(text="⏹ Остановить таймер", callback_data=f"start_timer_{book_id}")],
                [InlineKeyboardButton(text="✅ Книга прочитана", callback_data=f"mark_read_{book_id}")],
                [InlineKeyboardButton(text="⬅ Назад", callback_data="back_to_catalog")]
            ])
            try:
                await callback.message.edit_reply_markup(reply_markup)
            except Exception:
                pass

@dp.callback_query(F.data == "back_to_start")
async def back_to_start(callback: types.CallbackQuery, state: FSMContext):
    data = await state.get_data()

    # 1) Удаляем сообщение книги, если оно осталось
    book_message_id = data.get("book_message_id")
    if book_message_id:
        try:
            await bot.delete_message(callback.from_user.id, book_message_id)
        except Exception:
            pass
        await state.update_data(book_message_id=None)

    # 2) Удаляем сообщение каталога, если оно было сохранено отдельно
    catalog_message_id = data.get("catalog_message_id")
    if catalog_message_id:
        try:
            await bot.delete_message(callback.from_user.id, catalog_message_id)
        except Exception:
            pass
        await state.update_data(catalog_message_id=None)

    # 3) Если текущее callback.message — не главное (например, это сообщение каталога),
    #    удаляем его, чтобы под стартом не оставалось лишнего сообщения.
    main_message_id = data.get("main_message_id")
    try:
        if callback.message and callback.message.message_id != main_message_id:
            await bot.delete_message(callback.from_user.id, callback.message.message_id)
    except Exception:
        pass

    # 4) Формируем стартовый текст и путь до логотипа
    logo_path = "970111763390b956fad9089dd2cefcf0.jpg"
    caption = (
        "📚 Привет, читатель!\n\n"
        "Добро пожаловать в *BookBot* 📖\n\n"
        "Выбери действие ниже:"
    )

    # 5) Пытаемся отредактировать существующее главное сообщение (если есть)
    if main_message_id:
        try:
            if os.path.exists(logo_path):
                await bot.edit_message_media(
                    chat_id=callback.from_user.id,
                    message_id=main_message_id,
                    media=types.InputMediaPhoto(
                        media=FSInputFile(logo_path),
                        caption=caption,
                        parse_mode="Markdown"
                    ),
                    reply_markup=start_keyboard()
                )
            else:
                await bot.edit_message_text(
                    chat_id=callback.from_user.id,
                    message_id=main_message_id,
                    text=caption,
                    reply_markup=start_keyboard(),
                    parse_mode="Markdown"
                )
            # обновляем состояние
            await state.update_data(current="start")
            return
        except TelegramBadRequest:
            # если редактировать нельзя (например, сообщение удалено) — пойдём дальше и отправим новое
            pass

    # 6) Если мы не отредактировали старое (или его нет) — отправляем новое стартовое сообщение
    if os.path.exists(logo_path):
        sent = await callback.message.answer_photo(
            photo=FSInputFile(logo_path),
            caption=caption,
            reply_markup=start_keyboard(),
            parse_mode="Markdown"
        )
    else:
        sent = await callback.message.answer(
            caption,
            reply_markup=start_keyboard(),
            parse_mode="Markdown"
        )

    # 7) Сохраняем ID нового стартового сообщения
    await state.update_data(main_message_id=sent.message_id, current="start")

@dp.callback_query(F.data == "back_to_catalog")
async def back_to_catalog(callback: types.CallbackQuery, state: FSMContext):
    data = await state.get_data()
    current = data.get("current")

    # Удаляем сообщение книги
    book_message_id = data.get("book_message_id")
    if book_message_id:
        try:
            await bot.delete_message(callback.from_user.id, book_message_id)
        except Exception:
            pass
        await state.update_data(book_message_id=None)

    # Восстанавливаем сообщение каталога
    catalog_message_id = data.get("catalog_message_id")
    if catalog_message_id:
        async with aiosqlite.connect(DB_PATH) as db:
            cur = await db.execute("SELECT id, title FROM books")
            books = await cur.fetchall()

        text = "📚 *Каталог книг:*\n\n"
        if not books:
            text += "Пока что здесь пусто 😅"
        keyboard = [
            [InlineKeyboardButton(text=title, callback_data=f"book_{book_id}")]
            for book_id, title in books
        ]
        keyboard.append([InlineKeyboardButton(text="⬅ Назад", callback_data="back_to_start")])
        markup = InlineKeyboardMarkup(inline_keyboard=keyboard)

        await bot.send_message(callback.from_user.id, text=text, reply_markup=markup, parse_mode="Markdown")
        await state.update_data(current="catalog", catalog_message_id=None)
